GET /health

Scope: read:health

curl https://backend.goodos.app/api/v1/health \
  -H "X-GoodOS-API-Key: $GOODOS_API_KEY"

GET /apps

Scope: read:apps

curl https://backend.goodos.app/api/v1/apps \
  -H "X-GoodOS-API-Key: $GOODOS_API_KEY"

GET /storage/buckets

Scope: read:storage

curl https://backend.goodos.app/api/v1/storage/buckets \
  -H "X-GoodOS-API-Key: $GOODOS_API_KEY"

GET /storage/files

Scope: read:storage

curl https://backend.goodos.app/api/v1/storage/files \
  -H "X-GoodOS-API-Key: $GOODOS_API_KEY"

POST /functions/:slug

Scope: execute:functions

curl -X POST https://backend.goodos.app/api/v1/functions/health-check \
  -H "Content-Type: application/json" \
  -H "X-GoodOS-API-Key: $GOODOS_API_KEY" \
  --data '{"source":"api-docs","message":"Testing function call"}'
